### Importations des csv dans mysql .

LOAD DATA INFILE '/var/lib/mysql-files/cet_communes.csv'
INTO TABLE cet_communes
FIELDS TERMINATED BY ','
ENCLOSED BY '"'
LINES TERMINATED BY '\n'
IGNORE 1 ROWS;
